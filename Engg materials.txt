    let currentBranch = null;

    // ===== NAVIGATION =====
    function showPage(id) {
      document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
      document.getElementById(id).classList.add('active');
    }

    function showProfile() { showPage('profilePage'); }

    function showBranches() {
      const list = document.getElementById('branchList');
      list.innerHTML = branches.map(b => `
        <button class="branch-card" data-branch="${b.id}" onclick="showSemesters('${b.id}')">
          <span class="emoji">${b.icon}</span>
          <div class="info">
            <div class="short">${b.shortName}</div>
            <div class="full">${b.name}</div>
          </div>
          <svg class="chevron icon-svg" viewBox="0 0 24 24"><path d="M9 5l7 7-7 7"/></svg>
        </button>
      `).join('');
      showPage('branchPage');
    }

    function showSemesters(branchId) {
      currentBranch = branches.find(b => b.id === branchId);
      const sems = semesterSubjects[branchId] || [];
      document.getElementById('semIcon').textContent = currentBranch.icon;
      document.getElementById('semTitle').textContent = currentBranch.shortName;
      document.getElementById('semesterGrid').innerHTML = sems.map(s => `
        <button class="sem-card" onclick="showSubjects('${branchId}', ${s.number})">
          <span class="sem-num">${s.number}</span>
          <span class="sem-label">Semester</span>
        </button>
      `).join('');
      showPage('semesterPage');
    }

    function showSubjects(branchId, semNum) {
      currentBranch = branches.find(b => b.id === branchId);
      const sem = (semesterSubjects[branchId] || []).find(s => s.number === semNum);
      document.getElementById('subIcon').textContent = currentBranch.icon;
      document.getElementById('subTitle').textContent = `${currentBranch.shortName} — Semester ${semNum}`;
      document.getElementById('subjectBackBtn').onclick = () => showSemesters(branchId);

      const fileIcon = `<svg class="icon-svg" viewBox="0 0 24 24"><path d="M14 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V8z"/><polyline points="14 2 14 8 20 8"/><line x1="16" y1="13" x2="8" y2="13"/><line x1="16" y1="17" x2="8" y2="17"/><polyline points="10 9 9 9 8 9"/></svg>`;
      const extIcon = `<svg class="icon-svg" style="width:16px;height:16px" viewBox="0 0 24 24"><path d="M18 13v6a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2V8a2 2 0 0 1 2-2h6"/><polyline points="15 3 21 3 21 9"/><line x1="10" y1="14" x2="21" y2="3"/></svg>`;

      document.getElementById('subjectList').innerHTML = (sem?.subjects || []).map(sub => {
        const has = !!sub.materialUrl;
        const inner = `
          <div class="subject-icon ${has ? 'active' : 'inactive'}">${fileIcon}</div>
          <div class="subject-info">
            <div class="subject-name">${sub.name}</div>
            <div class="subject-code">${sub.code}</div>
          </div>
          ${has ? `<span class="view-pdf">View PDF ${extIcon}</span>` : `<span class="coming-soon">Coming soon</span>`}
        `;
        return has
          ? `<a href="${sub.materialUrl}" target="_blank" rel="noopener noreferrer" class="subject-card has-material">${inner}</a>`
          : `<div class="subject-card">${inner}</div>`;
      }).join('');
      showPage('subjectPage');
    }
